196void SurviveMode::manageMobSpawn() {
197 // Vérifie le délai entre plusieurs apparitions d'ennemie
198 auto spawnBasic = std::chrono::system_clock::now();
199 std::chrono::duration<double> elapsed_seconds_basic = spawnBasic-m_rateBetweenBasicSpawn;
200
201 // Crée un délai entre plusieurs apparitions
202 if (elapsed_seconds_basic.count() > m_spawnRate) {
203 int additionnalSpawnByTime = m_timeMinuteAlive > 2 ? m_timeMinuteAlive -2 : 0;
204 spawnBasicEnemy(1 + additionnalSpawnByTime);
205 m_rateBetweenBasicSpawn = std::chrono::system_clock::now();
206 }
207
208 // Fait apparaître un boss toutes les 30 secondes
209 if ((m_timeSecondeAlive == 0 || m_timeSecondeAlive == 30) && m_timeMinuteAlive > 0 && m_bossCanSpawn) {
210 m_bossCanSpawn = false;
211 spawnBossEnemy();
212 }
213
214 // Fait apparâitre un coffre toutes les minutes
215 if (m_needChest) {
216 m_needChest = false;
217 Chest* pChest = new Chest();
218 connect(pChest, &Chest::PlayerTakeChest, this, &SurviveMode::newLevelForPlayer);
219
220 m_pScene->addSpriteToScene(pChest);
221 pChest->registerForTick();
222

228void SurviveMode::updateGameTime() {
229 // Mets à jour le temps de vie
230 auto now = std::chrono::system_clock::now();
231 std::chrono::duration<double> seconds = now-m_rateSeconde;
232 if (seconds.count() > 1) {
233 if (m_timeSecondeAlive >= 59) {
234 m_timeMinuteAlive++;
235 m_timeSecondeAlive = -1;
236 }
237 m_timeSecondeAlive++;
238
239 if (m_timeSecondeAlive == 5 || m_timeSecondeAlive == 35) {
240 m_bossCanSpawn = true;
241 }
242
243 clearStat();
244 m_rateSeconde = std::chrono::system_clock::now();
245 }

347int SurviveMode::choiceRandomNumber() {
348 int bonusNumber = 9;
349 int bonus = reduceChance(bonusNumber);
350
351 // Empêche les bonus qui ne sont plus activables de revenir
352 if (m_autoFireOn) {
353 do {
354 bonus = reduceChance(bonusNumber);
355 } while (bonus == 1);
356 }
357 if (m_pPlayer->getMultiFireStatu()) {
358 int saveAutoFire = m_autoFireOn ? 1 : -1;
359 do {
360 bonus = reduceChance(bonusNumber);
361 } while (bonus == 6 || bonus == saveAutoFire);
362 }
363 if (m_pPlayer->getShootingRate() <= 0.3) {
364 int saveAutoFire = m_autoFireOn ? 1 : -1;
365 int saveMultiFire = m_pPlayer->getMultiFireStatu() ? 6 : -1;
366 do {
367 bonus = reduceChance(bonusNumber);
368 } while (bonus == 2 || bonus == saveAutoFire || bonus == saveMultiFire);
369 }
370
371 return bonus;
372}

376int SurviveMode::reduceChance(int bonusNumber) {
377 int bonus = rand() % bonusNumber;
378
379 // Réduis un peu les chances d'obtenir le bonus 1
380 if (bonus == 1) {
381 if (rand() % 2 == 0)
382 bonus = rand() % bonusNumber;
383 }
384
385 // Réduis les chances d'obtenir le bonus 0
386 if (bonus == 0)
387 bonus = rand() % bonusNumber;
388
389 // Réduis les chances d'obtenir le bonus 6
390 if (bonus == 6) {
391 bonus = rand() % bonusNumber;
392 // réduir encore plus les chances avant le niveau 8 du joueur
393 if (m_pPlayer->getPlayerLevel() < 8 && bonus == 6) {
394 bonus = rand() % bonusNumber;
395 }
396 }
397
398 // Réduis les chances d'obtenir le bonus 7
399 if (bonus == 7) {
400 bonus = rand() % bonusNumber;
401 // Réduis encore plus les chances avant le niveau 10 du joueur
402 if (m_pPlayer->getPlayerLevel() < 10 && bonus == 7) {
403 bonus = rand() % bonusNumber;
404 }
405 }
406
407 return bonus;
408}

429void SurviveMode::applyBonus(int bonusIndex) {
430 switch (bonusIndex) {
431 case 0:
432 // NUKE, tue toutes les ennemies sur le terrain
433 m_numberOfBombs++;
434 m_pHudTextBombe->setPlainText(QString::number(m_numberOfBombs));
435 break;
436 case 1:
437 // AUTO_FIRE, mets en place un tir automatique
438 m_autoFireOn = true;
439 m_pPlayer->reduceAttackSpeed(0.1);
440 break;
441 case 2:
442 // ATTACK_SPEED, améliore la cadence de tir du vaisseau
443 m_pPlayer->reduceAttackSpeed(0.1);
444 m_pPlayer->setCritDamage(m_pPlayer->getCritDamage() + 0.2);
445 break;
446 case 3:
447 // HEALTH_POINT, améliore les points de vie du vaisseau
448 m_pPlayer->setMaximumHealthPoint(m_pPlayer->getMaximumHealthPoint() + 10);
449 m_pPlayer->setActualHealthPoint(m_pPlayer->getMaximumHealthPoint());
450 m_pPlayer->setHealthBarMaximum(m_pPlayer->getMaximumHealthPoint());
451 m_pPlayer->setHealthBarValue(m_pPlayer->getMaximumHealthPoint());
452 m_pPlayer->setDamageReduction(m_pPlayer->getDamageReduction() + 1);
453 break;
454 case 4:
455 // BULLET_DAMAGE, améliore les dégâts des balles
456 m_pPlayer->setBulletDamage(m_pPlayer->getBulletDamage() + 20);
457 break;
458 case 5:
459 // BULLET_PIERCING, augmente le percement des balles
460 m_pPlayer->setBulletPierce(m_pPlayer->getBulletPierce() +1);
461 m_pPlayer->setLessDamagePiercing(m_pPlayer->getLessDamagePiercing() - 0.2);
462 break;
463 case 6:
464 // MULTI_FIRE, tire deux balles supplémentaires
465 m_pPlayer->setMultiFireStatu(true);
466 break;
467 case 7:
468 // ALL_BONUS, applique tous les bonus
469 m_pPlayer->reduceAttackSpeed(0.1);
470 m_pPlayer->setMaximumHealthPoint(m_pPlayer->getMaximumHealthPoint() + 10);
471 m_pPlayer->setActualHealthPoint(m_pPlayer->getMaximumHealthPoint());
472 m_pPlayer->setHealthBarMaximum(m_pPlayer->getMaximumHealthPoint());
473 m_pPlayer->setHealthBarValue(m_pPlayer->getMaximumHealthPoint());
474 m_pPlayer->setBulletDamage(m_pPlayer->getBulletDamage() + 20);
475 m_pPlayer->setBulletPierce(m_pPlayer->getBulletPierce() +1);
476 m_pPlayer->setShipSpeed(m_pPlayer->getShipSpeed() + 50);
477 m_pPlayer->setLessDamagePiercing(m_pPlayer->getLessDamagePiercing() - 0.2);
478 m_autoFireOn = true;
479 m_pPlayer->setMultiFireStatu(true);
480 m_pPlayer->setDamageReduction(m_pPlayer->getDamageReduction() + 1);
481 m_pPlayer->setCritDamage(m_pPlayer->getCritDamage() + 0.2);
482 break;
483 case 8:
484 // SHIP_SPEED, améliore la vitesse du vaisseau
485 m_pPlayer->setShipSpeed(m_pPlayer->getShipSpeed() + 50);
486 }
487
488 m_waitingForChoice = false;
489}

532void SurviveMode::keyPressed(int key) {

534
535 int movement = 0;
