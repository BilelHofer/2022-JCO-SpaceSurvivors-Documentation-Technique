128void SurviveMode::tick(long long elapsedTimeInMilliseconds) {
129 // met à jour l'HUD
130 m_pHudTextTime->setPlainText(QString::number(m_timeMinuteAlive) + "m " + QString::number(m_timeSecondeAlive) + "s");
131
132 // Regarde le mode de jeux
133 if (m_gameMode == RUNNING) {
134
135 if (m_nukeNeedLaunch) {
136 m_nukeNeedLaunch = false;
137 emit nukeIsLaunched();
138 m_numberOfBombs--;
139 m_pHudTextBombe->setPlainText(QString::number(m_numberOfBombs));
140 }
141
142 if (m_autoFireOn) {
143 auto fireNow = std::chrono::system_clock::now();
144 std::chrono::duration<double> elapsed_seconds = fireNow-m_rateBetweenShot;
145
146 // crée un délai entre plusieur tire
147 if (elapsed_seconds.count() > m_pPlayer->m_shootingRate) {
148 m_pPlayer->shoot(m_pScene);
149
150 m_rateBetweenShot = std::chrono::system_clock::now();
151 }
152 }
153
154 // Met à jour le temps de vie
155 auto now = std::chrono::system_clock::now();
156 std::chrono::duration<double> seconds = now-m_rateSeconde;
157 if (seconds.count() > 1) {
158 if (m_timeSecondeAlive == 59) {
159 m_timeMinuteAlive++;
160 m_timeSecondeAlive = -1;
161 }
162 m_timeSecondeAlive++;
163 clearStat();
164 m_rateSeconde = std::chrono::system_clock::now();
165 }
166
167 // vérifi le delay entre plusieur spawn d'ennemie
168 auto spawnBasic = std::chrono::system_clock::now();
169 std::chrono::duration<double> elapsed_seconds_basic = spawnBasic-m_rateBetweenBasicSpawn;
170
171 // crée un délai entre plusieur spawn
172 if (elapsed_seconds_basic.count() > m_spawnRate) {
173 int additionnalSpawnByTime = m_timeMinuteAlive > 2 ? m_timeMinuteAlive -2 : 0;
174 spawnBasicEnemy(1 + additionnalSpawnByTime);
175 m_rateBetweenBasicSpawn = std::chrono::system_clock::now();
176 }
177
178 // fait spawn un boss toutes les 30 seconde
179 if ((m_timeSecondeAlive == 0 || m_timeSecondeAlive == 30) && m_timeMinuteAlive > 0 && m_pScene->m_bossIsAlive == false) {
180 spawnBossEnemy();
181 m_pScene->m_bossIsAlive = true;
182 }
183
184 // Détecte si le joueur est en mouvement.
185 if (m_playerAxeX != 0 || m_playerAxeY != 0) {
186 movePlayer(elapsedTimeInMilliseconds);
187 }
188 } else if (m_gameMode == BREAK || m_gameMode == ENDED) {
189 // change l'image du boutton au hover
190 if (ToolBox().isInsideSprite(m_pButtonMenu, m_lastMousePos)) {
191 m_pButtonMenu->setCurrentAnimationFrame(1);
