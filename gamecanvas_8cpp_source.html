196bool GameCanvas::eventFilter(QObject* pWatched, QEvent* pEvent)
197{
198 // Les événements claviers doivent être traités au niveau de GameView, sinon, certaines touches
199 // auront un effet indésirable pris en charge par QGraphicsView (ou plus exactement QAbstractScrollBar, qui
200 // gère le comportement des touches fléchées et de PageUp et PageDown).
201 // Pour que l'événement ne soit plus traité plus loin (et donc soit absorbé ici), on retourne true, comme
202 // expliqué dans la doc de QObject::eventFilter() :
203 //
204 // "In your reimplementation of this function, if you want to filter the event out, i.e. stop it being handled
205 // further, return true; otherwise return false."
206 //
207 if (qobject_cast<QGraphicsView*>(pWatched)) {
208 switch (pEvent->type()) {
209 case QEvent::KeyPress: this->keyPressed(static_cast<QKeyEvent*>(pEvent)); return true;
210 case QEvent::KeyRelease: this->keyReleased(static_cast<QKeyEvent *>(pEvent)); return true;
211 default : return QObject::eventFilter(pWatched, pEvent);
212 }
213 }
214
215 // Les événements restants proviennent de GameScene.
216 // Attention : les événements de la souris, une fois traités par GameCanevas, doivent pouvoir être gérés plus
217 // loin, par exemple dans le cas où la scène intègre des QWidget. Il faut donc retourner false.
218 switch (pEvent->type()) {
219 case QEvent::GraphicsSceneMouseMove: this->mouseMoved(static_cast<QGraphicsSceneMouseEvent*>(pEvent)); return false;
220 case QEvent::GraphicsSceneMousePress: this->mouseButtonPressed(static_cast<QGraphicsSceneMouseEvent*>(pEvent)); return false;
221 case QEvent::GraphicsSceneMouseRelease: this->mouseButtonReleased(static_cast<QGraphicsSceneMouseEvent*>(pEvent)); return false;
222 default : return QObject::eventFilter(pWatched, pEvent);
223 }
224}

238void GameCanvas::keyPressed(QKeyEvent* pKeyEvent) {
239 // Supprimer ce premier test si la répétition de touche doit être signalée.
240 if (pKeyEvent->isAutoRepeat())
241 pKeyEvent->ignore();
242 else {
243 if (m_actualCore == GAME) {
244 m_pGameCore->keyPressed(pKeyEvent->key());
245 } else if (m_actualCore == MENU){
246 m_pMenuCore->keyPressed(pKeyEvent->key());
247 }
248
249 if (pKeyEvent->modifiers()==(Qt::ShiftModifier|Qt::ControlModifier)) {
250 switch (pKeyEvent->key()) {
251 case Qt::Key_I:
252 if (m_pDetailedInfosItem)
253 m_pDetailedInfosItem->setVisible(!m_pDetailedInfosItem->isVisible());
254 break;
255 case Qt::Key_P:
256 m_tickTimer.setInterval(m_tickTimer.interval()+1);
257 qDebug() << "Tick interval set to " << m_tickTimer.interval();
258 break;
259 case Qt::Key_M:
260 m_tickTimer.setInterval(m_tickTimer.interval()-1);
261 qDebug() << "Tick interval set to " << m_tickTimer.interval();
262 break;
263 }
264 }
265 pKeyEvent->accept();
266 }
267}

329void GameCanvas::onTick() {
330 long long elapsedTime = m_lastUpdateTime.elapsed();
331
332 // On évite une division par zéro (peu probable, mais on sait jamais)
333 if (elapsedTime < 1)
334 elapsedTime = 1;
335
336 m_lastUpdateTime.start();
337
338 if (m_actualCore == GAME) {
339 m_pGameCore->tick(elapsedTime);
340 } else if (m_actualCore == MENU) {
341 m_pMenuCore->tick(elapsedTime);
342 }
343
344 currentScene()->tick(elapsedTime);
345
346 if (m_pDetailedInfosItem && m_pDetailedInfosItem->isVisible())
347 m_pDetailedInfosItem->setPlainText(QString("FPS : %1, Elapsed : %2ms, Tick duration : %3ms")
348 .arg(1000/elapsedTime)
349 .arg(elapsedTime)
350 .arg(m_lastUpdateTime.elapsed()));
351
352 if (m_keepTicking)
353 m_tickTimer.start();
354}
