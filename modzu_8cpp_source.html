367void Modzu::replaceZone() {
368 // Replace la zone pour la bombe.
369 if (m_poisonBombZone != nullptr) {
370 // Mets à jour la zone autour du joueur.
371 m_poisonBombZone->setPos(m_playerPos.x() - m_poisonBombZone->width()/2,
372 m_playerPos.y() - m_poisonBombZone->height()/2);
373
374 // Teste si la zone sort de la map.
375 QRectF zoneRect = QRectF(m_poisonBombZone->x(), m_poisonBombZone->y(), m_poisonBombZone->width(), m_poisonBombZone->height());
376
377 if (!m_mapEdge.contains(zoneRect.toRect())) {
378 // Replacement vers la droite.
379 if (zoneRect.x() < m_mapEdge.x()) {
380 zoneRect.setX(m_mapEdge.x());
381 }
382 // Replacement vers la gauche.
383 if (zoneRect.x() + zoneRect.width() > m_mapEdge.x() + m_mapEdge.width()) {
384 zoneRect.setX(m_mapEdge.x() + m_mapEdge.width() - zoneRect.width());
385 }
386 // Replacement vers le bas.
387 if (zoneRect.y() < m_mapEdge.y()) {
388 zoneRect.setY(m_mapEdge.y());
389 }
390 // Remplacement vers le haut.
391 if (zoneRect.y() + zoneRect.height() > m_mapEdge.y() + m_mapEdge.height()) {
392 zoneRect.setY(m_mapEdge.y() + m_mapEdge.height() - zoneRect.height());
393 }
394 // Replace la zone.
395 m_poisonBombZone->setX(zoneRect.x());
396 m_poisonBombZone->setY(zoneRect.y());
397 }
398 // Déplace la zone devant le joueur.
399 m_poisonBombZone->setX(m_poisonBombZone->x() + m_pPlayer->getDirections().x()*120);
400 m_poisonBombZone->setY(m_poisonBombZone->y() + m_pPlayer->getDirections().y()*120);
401 }
402
403 // Replace la zone pour le laser.
404 if (m_laserZone != nullptr) {
405 // Mets à jour la zone autour du joueur.
406 m_laserZone->setPos(m_playerPos.x() - m_laserZone->width()/2,
407 m_playerPos.y() - m_laserZone->height()/2);
408
409 // Ce place dans la direction du joueur.
410 m_laserZone->setX(m_laserZone->x() + m_pPlayer->getDirections().x()*(m_pPlayer->getShipSpeed()*1.25));
411
412 // Si le joueur ne bouge pas le place sur le joueur.
413 if (m_pPlayer->getDirections().x() == 0) {
414 m_laserZone->setX(m_playerPos.x() - m_laserZone->width()/2);
415 }
416
417 // Si le laser sort de la zone le met au bord.
418 if (m_laserZone->x() < m_mapEdge.x()) {
419 m_laserZone->setX(m_mapEdge.x());
420 }
421 if (m_laserZone->x() + m_laserZone->width() > m_mapEdge.x() + m_mapEdge.width()) {
422 m_laserZone->setX(m_mapEdge.x() + m_mapEdge.width() - m_laserZone->width());
423 }
424 }
