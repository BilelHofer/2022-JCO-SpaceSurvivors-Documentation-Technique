50void PoisonBomb::tick(long long elapsedTimeInMilliseconds)

70 // Vérifie s’il y a une collision avec le joueur
71 if (dynamic_cast<Player*>(collision) != nullptr) {
72 m_pPlayer = dynamic_cast<Player*>(collision);
73
74 // Réduis la vitesse du joueur
75 if (!m_speedAlreadyDecrease) {
76 m_speedInLess = m_pPlayer->getShipSpeed() / 100 * m_percentOfSlowness;
77 m_pPlayer->setShipSpeed(m_pPlayer->getShipSpeed() - m_speedInLess);
78 m_speedAlreadyDecrease = true;

84 m_zoneDamageRate = 1000;

86 playerTouched = true;

90 // Annule la réduction de vitesse

92 if (m_speedAlreadyDecrease) {
93 m_pPlayer->setShipSpeed(m_pPlayer->getShipSpeed() + m_speedInLess);
