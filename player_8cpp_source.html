87void Player::damageTaken(int damage) {
88 auto damagedNow = std::chrono::system_clock::now();
89 std::chrono::duration<double> elapsed_seconds = damagedNow-m_rateBetweenTwoDamage;
90
91 // Crée un délai entre plusieurs dégâts subis.
92 if (elapsed_seconds.count() > m_damageRate) {
93 // Limite la réduction de dégâts pour toujours subir au moins 1 de dégâts.
94 damage = damage <= m_damageReduction ? 1 : damage - m_damageReduction;
95 m_actualHealthPoint -= damage;
96
97 // Si le joueur est encore en vie, mets à jour la barre de point de vie.
98 if (m_actualHealthPoint > 0)
99 m_pHealthBar->setValue(m_actualHealthPoint);
100
101 // Lance l'animation de prise de dégâts.
102 this->startAnimation();
103
104 // Vérifie que le joueur est toujours en vie

106 m_pHealthBar->setValue(0);
107 m_actualHealthPoint = 0;
108 emit playerDead();
109 m_playerIsDead = true;
110 }
