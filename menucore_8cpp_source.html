203
204 switch (modeActual) {

207 if (ToolBox().isInsideSprite(m_pButtonPlay, m_mousePos)) {
208 modeActual = MODE;

219 m_pGameCanvas->setCurrentScene(m_pSceneInfo);

221
222 // au clic revient à la scene du menu
223 if (ToolBox().isInsideSprite(m_pButtonBackInfo, m_mousePos)) {
224 m_pGameCanvas->setCurrentScene(m_pSceneMenu);
225 }
226 break;
227 case MODE:
228 // mode 1
229 if (ToolBox().isInsideSprite(m_pMode1, m_mousePos)) {
230 emit needChangeCore(1);

234 if (ToolBox().isInsideSprite(m_pMode2, m_mousePos)) {
235 emit needChangeCore(2);

237

239 if (ToolBox().isInsideSprite(m_pButtonBack, m_mousePos)) {
240 modeActual = MENU;
