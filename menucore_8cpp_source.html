337void MenuCore::getJson() {
338 gamesList.clear();
339 gameListBeat.clear();
340 gameListSurvive.clear();
341
342 // Récupère le fichier pour le mode survivre
343 QFile fileSurvive(GameFramework::resourcesPath() + QDir::separator() + QString("sauvegardes") + QDir::separator() + "survivre.json");
344
345 // ouvre le fichier
346 if (fileSurvive.open(QIODevice::ReadOnly)) {
347 QByteArray bytes = fileSurvive.readAll();
348 fileSurvive.close();
349
350 QJsonDocument document = QJsonDocument::fromJson(bytes);
351
352 // parse des datas
353 foreach(const QJsonValue & val, document.array()) {
354 double kill = val.toObject().value("kill").toDouble();
355 unsigned int damage = val.toObject().value("damage").toInt();
356 double minAlive = val.toObject().value("minuteAlive").toDouble();
357 double secAlive = val.toObject().value("secondeAlive").toDouble();
358 gamesList.append(Game(kill, damage, minAlive, secAlive));
359 }
360 }
361
362 // Récupère le fichier pour le mode battre
363 QFile fileBeat(GameFramework::resourcesPath() + QDir::separator() + QString("sauvegardes") + QDir::separator() + "battre.json");
364
365 // ouvre le fichier
366 if (fileBeat.open(QIODevice::ReadOnly)) {
367 QByteArray bytes = fileBeat.readAll();
368 fileBeat.close();
369
370 QJsonDocument document = QJsonDocument::fromJson(bytes);
371
372 // parse des datas
373 foreach(const QJsonValue & val, document.array()) {
374 double minAlive = val.toObject().value("minuteAlive").toDouble();
375 double secAlive = val.toObject().value("secondeAlive").toDouble();
376 double health = val.toObject().value("modzuPercentHealth").toDouble();
377 gamesList.append(Game(minAlive, secAlive, health));
378 }
379 }
380
381 // trie les listes
382 sortGameList();
383}

436QString MenuCore::getTotalTime() {
437 unsigned int minute = 0;
438 double totalSeconde = 0;
439
440 // Additionne le temps de toutes les parties.
441 for (int i = 0; i < gamesList.count(); i++) {
442 minute += gamesList.at(i).minAlive;
443 totalSeconde += gamesList.at(i).secAlive;
444 }
445
446 minute += totalSeconde / 60;
447
448 int nbMinuteAdditional = totalSeconde / 60.0;
449 double seconde = (totalSeconde / 60.0 - nbMinuteAdditional) * 60;
450
451 return QString::number(minute) + "m " + QString::number(seconde) + "s";
452}

467QString MenuCore::getBestPartSurvive() {
468 QList<Game> bestPart;
469 QString message ="Aucune partie jouée";
470
471 // Si aucune partie n'a été effectuée.
472 if (gameListSurvive.count() < 1) {
473 return message;
474
475 // Si une seul parti a été effectué.
476 } else if (gameListSurvive.count() == 1) {
477 bestPart.append(gameListSurvive.at(0));
478
479 // Trie le tableau pour ressortir les 3 meilleurs partis.
480 } else if (gameListSurvive.count() > 1) {
481 for (int i = 0; i < gameListSurvive.count() - 1; i++) {
482 for(int j = 0; j < gameListSurvive.count() - i - 1; j++) {
483
484 // Descends la plus grande valeur sur les minutes.
485 if (gameListSurvive.at(j).minAlive < gameListSurvive.at(j + 1).minAlive) {
486 gameListSurvive.swapItemsAt(j, j + 1);
487
488 // Si les minutes sont les mêmes descend la plus grande valeur sur les secondes.
489 } else if (gameListSurvive.at(j).minAlive == gameListSurvive.at(j + 1).minAlive) {
490 if (gameListSurvive.at(j).secAlive < gameListSurvive.at(j + 1).secAlive) {
491 gameListSurvive.swapItemsAt(j, j + 1);
492
493 // Si les secondes sont les mêmes descend la plus grande valeur sûre kill.
494 } else if (gameListSurvive.at(j).secAlive == gameListSurvive.at(j + 1).secAlive) {
495 if (gameListSurvive.at(j).kill < gameListSurvive.at(j + 1).kill) {
496 gameListSurvive.swapItemsAt(j, j + 1);
497 }
498 }
499 }
500 }
501 }
502
503 // Récupère les 3 meilleures parties.
504 bestPart.append(gameListSurvive.at(0));
505 if (gameListSurvive.count() > 1) {
506 bestPart.append(gameListSurvive.at(1));
507 if (gameListSurvive.count() > 2)
508 bestPart.append(gameListSurvive.at(2));
509 }
510 }
511
512 message ="Meilleures parties : \n";
513 for (auto part : bestPart) {
514 message +=
515 "Temps en vie : " + QString::number(part.minAlive) + "m " + QString::number(part.secAlive) + "s" +
516 "\nDégats infligé : " + QString::number(part.damage) +
517 "\nNombre d'ennemis tué : " + QString::number(part.kill) + "\n\n";
518 }
519
520 return message;
521}
