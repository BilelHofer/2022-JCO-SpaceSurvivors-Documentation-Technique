244void MenuCore::updateHotkey(int oldKey, int newKey) {
245 QFile file(GameFramework::resourcesPath() + QDir::separator() + QString("config") + QDir::separator() + "key.json"); // json file
246
247 bool error = false;
248 int keyIndex = 0;
249
250 arrKey[0] = m_pConfig->getUpKey();
251 arrKey[1] = m_pConfig->getDownKey();
252 arrKey[2] = m_pConfig->getRightKey();
253 arrKey[3] = m_pConfig->getLeftKey();
254 arrKey[4] = m_pConfig->getActiveKey();
255 arrKey[5] = m_pConfig->getBreakKey();
256
257 for (int i = 0; i < 6; i++) {
258 // annule le changement si la touche est déjà utilisé
259 if (arrKey[i] == newKey) {
260 error = true;
261
262 // enregistre la touche à modifier
263 } else if (arrKey[i] == oldKey) {
264 keyIndex = i;
265 }
266 }
267
268 if (!error) {
269
270 QString keyValue = "";
271 // fait correspondre la touche
272 switch (keyIndex) {
273 case 0:
274 keyValue = "up";
275 break;
276 case 1:
277 keyValue = "down";
278 break;
279 case 2:
280 keyValue = "right";
281 break;
282 case 3:
283 keyValue = "left";
284 break;
285 case 4:
286 keyValue = "active";
287 break;
288 case 5:
289 keyValue = "break";
290 }
291
292 // Crée la structure du Json
293 QJsonObject keyDetails = {
294 {"up", m_pConfig->getUpKey()},
295 {"down", m_pConfig->getDownKey()},
296 {"right", m_pConfig->getRightKey()},
297 {"left", m_pConfig->getLeftKey()},
298 {"active", m_pConfig->getActiveKey()},
299 {"break", m_pConfig->getBreakKey()}
300 };
301
302 // mets à jour la touche
303 keyDetails.erase(keyDetails.find(keyValue));
304 keyDetails.insert(keyValue, newKey);
305
306 // Crée un tableau Json
307 QJsonArray arrLog;
308 arrLog.push_back( keyDetails );
309
310 QJsonDocument doc( arrLog );
311
312 file.open( QIODevice::WriteOnly );
313
314 file.write(doc.toJson());
315 file.close();
316
317 m_pConfig->initVariable();
318 displaySettings();
319
320 } else {
321 displaySettings();
322
323
324 // Affiche Information
325 QGraphicsItem* pTextInfo = m_pSceneSettings->createText(QPointF(0 ,0), m_pConfig->getText("info_change_key_error1") + QKeySequence(newKey).toString().toUpper() + m_pConfig->getText("info_change_key_error2"), 30, Qt::white);

875 } else if (gameListSurvive.count() > 1) {
