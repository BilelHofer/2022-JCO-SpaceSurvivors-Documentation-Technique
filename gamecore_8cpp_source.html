93void GameCore::movePlayer(long long elapsedTimeInMilliseconds) {
94
95 m_playerDirectionX = m_playerAxeX;
96 m_playerDirectionY = m_playerAxeY;
97
98 m_pPlayer->setDirection(m_playerDirectionX, m_playerDirectionY);
99
100 // normalize
101 double mag = sqrt(pow(m_playerAxeX, 2) + pow(m_playerAxeY, 2));
102 m_playerDirectionX /= mag;
103 m_playerDirectionY /= mag;
104
105 double playerMovementX = m_playerDirectionX * m_pPlayer->getShipSpeed() * elapsedTimeInMilliseconds / 1000.0;
106 double playerMovementY = m_playerDirectionY * m_pPlayer->getShipSpeed() * elapsedTimeInMilliseconds / 1000.0;
107
108 // mise à jour de l'endroit de la hitbox
109 m_pPlayerHitBox->setPos(m_pPlayer->pos());
110
111 // Position du prochain emplacement X.
112 QPointF nextMoveX = QPointF(playerMovementX, 0);
113 QPointF nextMoveY = QPointF(0, playerMovementY);
114
115 QRectF nextBoundariesX = m_pPlayerHitBox->globalBoundingRect().translated(nextMoveX);
116 QRectF nextBoundariesY = m_pPlayerHitBox->globalBoundingRect().translated(nextMoveY);
117
118 // Déplace le joueur si il reste dans la map.
119 if (m_pMapEdge->contains(nextBoundariesX)) {
120 m_pPlayer->setX(m_pPlayer->x() + playerMovementX);
121 m_playerMoved = true;
122 }
123
124 if (m_pMapEdge->contains(nextBoundariesY)) {
125 m_pPlayer->setY(m_pPlayer->y() + playerMovementY);
126 m_playerMoved = true;
127 }
128
129 // Emit la nouvelle position du joueur, s’il a bougé.
130 if (m_playerMoved) {
131 emit notifyPlayerPosition(QPointF(m_pPlayer->x() + m_pPlayer->width()/2, m_pPlayer->y() + m_pPlayer->height()/2));
132 m_playerMoved = false;
133 }
134
135 // Mets à jour l'emplacement de la camera.
136 m_pScene->centerViewOn(m_pPlayer);
137
138 // Mets à jour la rotation du vaisseau.
139 shipLookMouse(m_lastMousePos);
140
141 // Mets à jour la position de la barre de vie et d'experience.
142 m_pPlayer->moveBar(m_pPlayer->x(), m_pPlayer->y() - 20);
143
144 emit notifyPlayerMoved(QPointF(m_pPlayer->x() + m_pPlayer->width()/2, m_pPlayer->y() + m_pPlayer->height()/2));
