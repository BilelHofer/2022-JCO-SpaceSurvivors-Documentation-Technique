404int GameCore::choiceRandomNumber() {
405 int bonusNumber = 9;
406 int bonus = reduceChance(bonusNumber);
407
408 // empêche les bonus qui ne sont plus activable de revenir
409 if (m_autoFireOn) {
410 do {
411 bonus = reduceChance(bonusNumber);
412 } while (bonus == 1);
413 }
414 if (m_pPlayer->m_multiFireActivated) {
415 int saveAutoFire = m_autoFireOn ? 1 : -1;
416 do {
417 bonus = reduceChance(bonusNumber);
418 } while (bonus == 6 || bonus == saveAutoFire);
419 }
420 if (m_pPlayer->m_shootingRate <= 0.3) {
421 int saveAutoFire = m_autoFireOn ? 1 : -1;
422 int saveMultiFire = m_pPlayer->m_multiFireActivated ? 6 : -1;
423 do {
424 bonus = reduceChance(bonusNumber);
425 } while (bonus == 2 || bonus == saveAutoFire || bonus == saveMultiFire);
426 }
427
428 return bonus;
429}

433int GameCore::reduceChance(int bonusNumber) {
434 int bonus = rand() % bonusNumber;
435
436 // réduit un peu les chance d'obtenir le bonus 1
437 if (bonus == 1) {
438 if (rand() % 2 == 0)
439 bonus = rand() % bonusNumber;
440 }
441
442 // réduit les chance d'obtenir le bonus 0
443 if (bonus == 0)
444 bonus = rand() % bonusNumber;
445
446 // réduit les chance d'obtenir le bonus 6
447 if (bonus == 6) {
448 bonus = rand() % bonusNumber;
449 // réduit encore plus les chance avant le niveau 8 du joueur
450 if (m_pPlayer->m_playerLevel < 8 && bonus == 6) {
451 bonus = rand() % bonusNumber;
452 }
453 }
454
455 // réduit les chande d'obtenir le bonus 7
456 if (bonus == 7) {
457 bonus = rand() % bonusNumber;
458 // réduit encore plus les chance avant le niveau 10 du joueur
459 if (m_pPlayer->m_playerLevel < 10 && bonus == 7) {
460 bonus = rand() % bonusNumber;
461 }
462 }
463
464 return bonus;

469void GameCore::applyBonus(int bonusIndex) {
470 switch (bonusIndex) {
471 case 0:
472 // NUKE, tue tous les ennemie sur le terrain, donne l'exp
473 m_numberOfBombs++;
474 m_pHudTextBombe->setPlainText(QString::number(m_numberOfBombs));
475 break;
476 case 1:
477 // AUTO_FIRE, met en place un tire automatique
478 m_autoFireOn = true;
479 m_pPlayer->reduceAttackSpeed(0.1);
480 break;
481 case 2:
482 // ATTACK_SPEED, améliore l'attaque speed du vaisseau
483 m_pPlayer->reduceAttackSpeed(0.1);
484 m_pPlayer->m_critDamage += 0.2;
485 break;
486 case 3:

488 m_pPlayer->m_maxHealtPoint += 10;
489 m_pPlayer->m_actualHealtPoint = m_pPlayer->m_maxHealtPoint;
490 m_pPlayer->m_pHealthBar->setMaximum(m_pPlayer->m_maxHealtPoint);
491 m_pPlayer->m_pHealthBar->setValue(m_pPlayer->m_actualHealtPoint);
492 m_pPlayer->m_damageReduction++;
493 break;
494 case 4:
495 // BULLET_DAMAGE, améliore les degats des balles
496 m_pPlayer->m_bulletDamage += 20;
497 break;
498 case 5:
499 // BULLET_PIERCING, augemente le percement des balles
500 m_pPlayer->m_bulletPierce += 1;
501 m_pPlayer->m_lessDamagePiercing -= 0.2;
502 break;
503 case 6:
504 // MULTI_FIRE, tire 2 ball supplémentaire
505 m_pPlayer->m_multiFireActivated = true;
506 break;
507 case 7:
508 // ALL_BONUS, applique tous les bonus
509 m_pPlayer->reduceAttackSpeed(0.1);
510 m_pPlayer->m_maxHealtPoint += 10;
511 m_pPlayer->m_actualHealtPoint = m_pPlayer->m_maxHealtPoint;
512 m_pPlayer->m_pHealthBar->setMaximum(m_pPlayer->m_maxHealtPoint);

516 m_pPlayer->m_shipSpeed += 50;
517 m_pPlayer->m_lessDamagePiercing -= 0.2;

519 m_pPlayer->m_multiFireActivated = true;
520 m_pPlayer->m_damageReduction++;
521 m_pPlayer->m_critDamage += 0.2;
522 break;
523 case 8:
524 // SHIP_SPEED, améliore la vitesse du vaisson
