390void Sprite::onNextAnimationFrame() {
391 if (m_animationList[m_currentAnimationIndex].isEmpty()) {
392 m_currentAnimationFrame = NO_CURRENT_FRAME;
393 return;
394 }
395
396 int PreviousAnimationFrame = m_currentAnimationFrame;
397 ++m_currentAnimationFrame;
398 if (m_currentAnimationFrame >= m_animationList[m_currentAnimationIndex].count()) {
399 m_currentAnimationFrame = 0;
400 if (m_emitSignalEOA)
401 emit animationFinished();
402 if (m_animationStopLater) {
403 m_animationStopLater = false;
404 stopAnimation(IMMEDIATE_STOP);
405 }
406 }
407 if (PreviousAnimationFrame != m_currentAnimationFrame) {
408 setPixmap(m_animationList[m_currentAnimationIndex][m_currentAnimationFrame]);
409 update();
410 }
411}
