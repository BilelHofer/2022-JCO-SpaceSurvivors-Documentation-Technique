125 // Regarde le mode de jeux
126 if (m_gameMode == RUNNING) {
127
128 if (m_autoFireOn) {
129 auto fireNow = std::chrono::system_clock::now();
130 std::chrono::duration<double> elapsed_seconds = fireNow-m_rateBetweenShot;
131
132 // crée un délai entre plusieur tire
133 if (elapsed_seconds.count() > m_pPlayer->m_shootingRate) {
134 m_pPlayer->shoot(m_pScene);
135
136 m_rateBetweenShot = std::chrono::system_clock::now();
137 }
138 }
139
140 // Met à jour le temps de vie
141 auto now = std::chrono::system_clock::now();
142 std::chrono::duration<double> seconds = now-m_rateSeconde;
143 if (seconds.count() > 1) {
144 if (m_timeSecondeAlive == 59) {
145 m_timeMinuteAlive++;
146 m_timeSecondeAlive = -1;
147 }
148 m_timeSecondeAlive++;
149 clearStat();
150 m_rateSeconde = std::chrono::system_clock::now();
151 }
152
153 // Détecte si le joueur est en mouvement.
154 if (m_playerAxeX != 0 || m_playerAxeY != 0) {
155 movePlayer(elapsedTimeInMilliseconds);
156 }
157 } else if (m_gameMode == BREAK || m_gameMode == ENDED) {
158 // change l'image du boutton au hover
159 if (ToolBox().isInsideSprite(m_pButtonMenu, m_lastMousePos)) {
160 m_pButtonMenu->setCurrentAnimationFrame(1);
