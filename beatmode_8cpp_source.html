128void BeatMode::tick(long long elapsedTimeInMilliseconds) {
129 // met à jour l'HUD
130 m_pHudTextTime->setPlainText(QString::number(m_timeMinuteAlive) + "m " + QString::number(m_timeSecondeAlive) + "s");
131
132 // Regarde le mode de jeux
133 if (m_pScene->getGameMode() == GameScene::RUNNING) {
134
135 // Mets à jour le temps de vie
136 auto now = std::chrono::system_clock::now();
137 std::chrono::duration<double> seconds = now-m_rateSeconde;
138 if (seconds.count() > 1) {
139 if (m_timeSecondeAlive == 59) {
140 m_timeMinuteAlive++;
141 m_timeSecondeAlive = -1;
142 }
143 m_timeSecondeAlive++;
144 clearStat();
145 m_rateSeconde = std::chrono::system_clock::now();
146 }
147
148 // si le shoot est activer tir
149 if (m_inShoot) {
150 auto fireNow = std::chrono::system_clock::now();
151 std::chrono::duration<double> elapsed_seconds = fireNow-m_rateBetweenShot;
152
153 // crée un délai entre plusieur tir
154 if (elapsed_seconds.count() > m_pPlayer->getShootingRate()) {
155 m_pPlayer->shoot();
156
157 m_rateBetweenShot = std::chrono::system_clock::now();
158 }
159 }
160
161 // Détecte si le joueur est en mouvement et s'il n'est pas déplacé de force.
162 if ((m_playerAxeX != 0 || m_playerAxeY != 0)) {
163 if (!m_playerIsMovedByForce)
164 movePlayer(elapsedTimeInMilliseconds);
165 } else {
166 m_pPlayer->setDirection(0, 0);
167 m_pModzu->replaceZone();
168 }
169 } else if (m_pScene->getGameMode() == GameScene::BREAK || m_pScene->getGameMode() == GameScene::ENDED) {
170 // change l'image du bouton au hover
171 if (ToolBox().isInsideSprite(m_pButtonMenu, m_lastMousePos)) {
172 m_pButtonMenu->setCurrentAnimationFrame(1);
173 } else {
174 m_pButtonMenu->setCurrentAnimationFrame(0);
175 m_pButtonMenu->setPos(m_pSceneBreak->width()/2 - m_pButtonMenu->width()/2, m_pSceneBreak->height() - m_pButtonMenu->height() - 20);
176 }
177 if (ToolBox().isInsideSprite(m_pButtonReplay, m_lastMousePos)) {
178 m_pButtonReplay->setCurrentAnimationFrame(1);
179 m_pButtonReplay->setPos(m_pButtonMenu->x() + m_pButtonMenu->width() + 20, m_pButtonMenu->y());
180 } else {
181 m_pButtonReplay->setCurrentAnimationFrame(0);
182 m_pButtonReplay->setPos(m_pButtonMenu->x() + m_pButtonMenu->width() + 20, m_pButtonMenu->y());
183 }
184 }
